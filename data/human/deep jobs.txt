# Deep Recovery and Rescue Jobs

# Easy ones
mission "Recovery Standard Deep Asset"
	name "Secure Deep Security Sensor Core in <destination>"
	description "Deep Security Intelligence requires a discreet captain to recover a classified black-box data tape weighting <cargo> tons from a wrecked vessel in the <destination> system. Be advised that anti-tamper security grids may be active. Payment for successful extraction is 45,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 2 7
		not attributes "uninhabited"
	cargo 10 20 0.7
	to offer
		"rr_intro_completed" == 1
	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Headhunter" "Classified Deep Wreck"
		on board
			conversation
				"Your system techs cut into the primary command node, pulling the black-box data tape. The tech reports the telemetry log is secured, but the drone's anti-tamper security grid has spiked the battery cells and it is going up."
	on complete
		"deep_asset_ambush_1_ready" = 1
		conversation
			"You return to <origin>. Deep Intelligence agents take custody of the drive under strict radio silence curtains and your holds are clear."
		"rr_operations_jobs_done" ++
		"reputation: Deep Security" += 0.01
		payment 45000
	on fail
		conversation
			"Deep Security Intelligence places a secure administrative lockdown black mark across your spacer pilot file."
		"reputation: Deep Security" -= 0.15

mission "Recovery Standard Deep Asset Ambush"
	invisible
	to offer
		"deep_asset_ambush_1_ready" == 1
	deadline 1
	npc

		personality "hunting"
		fleet "Small Southern Pirates"
	on complete
		"deep_asset_ambush_1_ready" = 0

# Hard ones
mission "Recovery Endgame Deep Security Core"
	name "CRITICAL: Salvage Deep Security Singularity Core in <destination>"
	description "CRITICAL CONTRACT: Extract a dangerously unstable singularity mechanism weighting <cargo> tons from a ruined Deep Security fortress in the <destination> system. The derelict's gravity containment is collapsing, and heavy raider warships are tracking the beacon. Failure will result in severe license penalties. Payment for this endgame salvage operation is 250,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 8 13
		not attributes "uninhabited"
	cargo 110 150 0.15
	to offer
		"your_mission_cooldown_date" <= day
		"rr_hard_tier_unlocked" == 1
	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Leviathan" "Ruined Deep Security Fortress"
		on board
			conversation
				"Your specialist crew extracts the dangerously unstable singularity mechanism. The chief yells that the core is secured but the gravity containment field is collapsing on the derelict, urging you to disengage as heavy raider warships and swarms of fast interceptors are dropping directly out of hyperspace on your coordinate tracks."
			"deep_endgame_salvage" = 1
	on complete
		"deep_endgame_core_ambush_ready" = 1
		conversation
			"You arrive safely at <origin>. Heavily armed deep operatives swarm your ship to silently extract the singularity core and your holds are clear."
		"rr_operations_jobs_done" ++
		"reputation: Deep Security" += 0.15
		"your_mission_cooldown_date" = day + 15
		payment 250000
	on fail
		conversation
			"Deep Security cuts all ties, completely revoking your clearance for losing a weapon of mass destruction."
		"reputation: Deep Security" -= 0.40
		"your_mission_cooldown_date" = day + 30
