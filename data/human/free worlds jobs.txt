# Free Worlds jobs

# Easy ones
mission "Recovery Standard Free Worlds Asset"
	name "Salvage Free Worlds Frontier Mineral Cache in <destination>"
	description "Free Worlds quartermasters need a ship to salvage a dense containment crate weighting <cargo> tons from a battered frontier miner in the <destination> system. Payment for returning the cache is 45,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 2 7
		not attributes "uninhabited"
	cargo 10 20 0.7
	to offer
		"rr_intro_completed" == 1
	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Bounder" "Battered Frontier Miner"
		on board
			conversation
				"Your crew winches the dense heavy containment crate into your primary loading bay."
	on complete
		"fw_asset_ambush_1_ready" = 1
		conversation
			"You return to <origin>. Free Worlds quartermasters take custody of the dense containment crate and your loading bay is clear."
		"rr_operations_jobs_done" ++
		"reputation: Free Worlds" += 0.01
		payment 45000
	on fail
		conversation
			"The Free Worlds militia command expresses their disappointment regarding the botched frontier recovery operation."
		"reputation: Free Worlds" -= 0.15

# Hard ones
mission "Recovery Endgame Free Worlds Crystals"
	name "CRITICAL: Salvage Free Worlds Plasma Cache in <destination>"
	description "CRITICAL CONTRACT: Extract a massive, highly volatile crystal cache weighting <cargo> tons from a crippled dreadnought in the <destination> system. This plasma cache is vital for stabilizing the local frontier power grid. Expect overwhelming pirate resistance. Payment for this endgame salvage operation is 250,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 8 13
		not attributes "uninhabited"
	cargo 110 150 0.15
	to offer
		"your_mission_cooldown_date" <= day
		"rr_hard_tier_unlocked" == 1
	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Leviathan" "Crippled Free Worlds Dreadnought"
		on board
			conversation
				"Your crew successfully extracts the massive highly volatile crystal crate."
	on complete
		"fw_endgame_crystals_ambush_ready" = 1
		conversation
			"You safely navigate down to <origin>. Specialized Free Worlds hazard teams carefully extract the highly volatile crystal cache. 'This stabilizes the local power grid for months, pilot,' the foreman says, 'You’ve done the frontier a great service today.'"
		"rr_operations_jobs_done" ++
		"reputation: Free Worlds" += 0.15
		"your_mission_cooldown_date" = day + 15
		payment 250000
	on fail
		conversation
			"The Free Worlds command officer sighs, the transmission crackling with static. 'That asset was vital to our defense of the frontier, pilot. You've failed us when we needed you most. You are grounded until further notice while we re-evaluate your clearance.'"
		"reputation: Free Worlds" -= 0.40
		"your_mission_cooldown_date" = day + 30
