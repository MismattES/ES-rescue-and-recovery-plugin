# Republic Recovery and Rescue jobs

# Easy ones
mission "Recovery Standard Republic Asset"
	name "Recover Republic Railgun Prototype in <destination>"
	description "The Republic Navy has contracted you to cut open the forward weapon bay of a stranded vessel in the <destination> system and extract its dense magnetic rails weighting <cargo> tons . Payment for returning the prototype is 45,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 2 7
		not attributes "uninhabited"
	cargo 10 20 0.7
	to offer
		"rr_intro_completed" == 1
	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Headhunter" "Stranded Navy Vessel"
		on board
			conversation
				"Your crew cuts open the forward weapon bay extracting the dense magnetic rails."
	on complete
		"rep_asset_ambush_1_ready" = 1
		conversation
			"You return to <origin>. Republic Navy quartermasters take custody of the magnetic rails and your holds are clear."
		"rr_operations_jobs_done" ++
		"reputation: Republic" += 0.01
		payment 45000
	on fail
		conversation
			"Republic Navy quartermasters note a significant failure on your operational record."
		"reputation: Republic" -= 0.15

# Hard ones
mission "Recovery Endgame Republic Core"
	name "CRITICAL: Salvage Republic Antimatter Engine Core in <destination>"
	description "CRITICAL CONTRACT: Use plasma-cutting frames to extract a massive, humming antimatter engine core weighting <cargo> tons from a burning Fleet Flagship in the <destination> system. This is a top-priority strategic initiative. Discretion and combat efficiency are mandatory. Payment for this endgame salvage operation is 250,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 8 13
		not attributes "uninhabited"
	cargo 110 150 0.15
	to offer
		"your_mission_cooldown_date" <= day
		"rr_hard_tier_unlocked" == 1
	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Leviathan" "Burning Fleet Flagship"
		on board
			conversation
				"Your plasma-cutting frames extract the massive humming magnetic containment vessel."
	on complete
		"rep_endgame_core_ambush_ready" = 1
		conversation
			"You safely navigate down to <origin>. Republic Fleet engineers stabilize the antimatter core as it is offloaded. 'This level of efficiency is exemplary,' the officer notes, 'The Republic will surely reward your discretion and skill.'"
		"rr_operations_jobs_done" ++
		"reputation: Republic" += 0.15
		"your_mission_cooldown_date" = day + 15
		payment 250000
	on fail
		conversation
			"A high-ranking Republic Admiral appears on your screen, his face cold and emotionless. 'The Republic does not tolerate incompetence. By losing that core, you have compromised a significant strategic initiative. Your flight privileges are revoked for the next thirty days.'"
		"reputation: Republic" -= 0.40
		"your_mission_cooldown_date" = day + 30
