# Syndicate Recovery and Rescue jobs

# Easy ones
mission "Recovery Standard Syndicate Asset"
	name "Recover Syndicate Cybernetic Core from <destination>"
	description "Syndicate operatives require an independent contractor to break terminal locks and disconnect a liquid-cooled processing stack weighting <cargo> tons from a disabled asset in the <destination> system. Payment for returning the cybernetic core is 45,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 2 7
		not attributes "uninhabited"
	cargo 10 20 0.7
	to offer
		"rr_intro_completed" == 1
	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Headhunter" "Disabled Syndicate Asset"
		on board
			conversation
				"Your team breaks the terminal locks disconnecting the liquid-cooled processing stack."
	on complete
		"syn_asset_ambush_1_ready" = 1
		conversation
			"You return to <origin>. Syndicate operatives silently collect the liquid-cooled processing stack and your holds are clear."
		"rr_operations_jobs_done" ++
		"reputation: Syndicate" += 0.01
		payment 45000
	on fail
		conversation
			"A Syndicate clerk efficiently logs your failure, noting a minor deduction in your reliability rating."
		"reputation: Syndicate" -= 0.15

mission "Recovery Standard Syndicate Asset Ambush"
	invisible

	deadline 1
	npc

	on complete
		"syn_asset_ambush_1_ready" = 0

# Hard ones
mission "Recovery Endgame Syndicate Tech"
	name "CRITICAL: Salvage Syndicate AI Logic Core in <destination>"
	description "CRITICAL CONTRACT: Override the cryogenic locks on a server farm aboard a ruined corporate super-freighter in the <destination> system. The Syndicate expects flawless execution and for you to ask no questions regarding the AI Logic Core. Payment for this endgame salvage operation is 250,000 credits."
	job
	repeat
	"offer precedence" 1000
	source
		not attributes "uninhabited"
	destination
		distance 8 13
		not attributes "uninhabited"
	cargo 110 150 0.15
	to offer
		"your_mission_cooldown_date" <= day
		"rr_hard_tier_unlocked" == 1
	npc
		government "Derelict"
		personality "derelict" "staying"
		ship "Leviathan" "Ruined Corporate Super-Freighter"
		on board
			conversation
				"Your specialist team overrides the cryogenic locks on the server farm."
	on complete
		"syn_endgame_tech_ambush_ready" = 1
		conversation
			"You safely navigate down to <origin>. Corporate executives supervise the extraction of the server farm. 'Flawless execution,' the supervisor notes, handing over your payment, 'The Syndicate appreciates assets who deliver results without asking questions.'"
		"rr_operations_jobs_done" ++
		"reputation: Syndicate" += 0.15
		"your_mission_cooldown_date" = day + 15
		payment 250000
	on fail
		conversation
			"A Syndicate black-suit supervisor leans into the camera, his voice a low, dangerous whisper. 'You were hired because we thought you were efficient. Losing that AI core has cost the Syndicate millions. Consider yourself suspended from all further operations while we settle our accounts.'"
		"reputation: Syndicate" -= 0.40
		"your_mission_cooldown_date" = day + 30
